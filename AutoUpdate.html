/** * AutoUpdateNotifier v1.0.0 * Bundle: UI + Logic + Firebase Listener */ class AutoUpdateNotifier { constructor(currentVersion, firestoreDb) { this.currentVersion = currentVersion; this.db = firestoreDb; this.injectStyles(); this.injectHTML(); this.initListener(); } injectStyles() { const style = document.createElement('style'); style.textContent = ` #update-notifier { position: fixed; bottom: 24px; left: 24px; z-index: 9999; transform: translateY(120%); opacity: 0; transition: all 0.6s cubic-bezier(0.23, 1, 0.32, 1); } #update-notifier.show { transform: translateY(0); opacity: 1; } .update-card { background: #09090b; border: 1px solid rgba(225, 29, 72, 0.5); padding: 20px; border-radius: 16px; display: flex; align-items: center; gap: 16px; box-shadow: 0 25px 50px -12px rgba(0, 0, 0, 0.5); max-width: 320px; color: white; font-family: sans-serif; } .update-icon { background: rgba(225, 29, 72, 0.1); color: #e11d48; padding: 12px; border-radius: 12px; } .update-btn { margin-top: 12px; background: #e11d48; color: white; font-size: 10px; font-weight: 900; text-transform: uppercase; padding: 8px 16px; border-radius: 8px; border: none; cursor: pointer; transition: background 0.2s; } .update-btn:hover { background: #be123c; } `; document.head.appendChild(style); } injectHTML() { const div = document.createElement('div'); div.id = 'update-notifier'; div.innerHTML = `

New Update Available

A new version is ready.

`; document.body.appendChild(div); } initListener() { this.db.collection('settings').doc('system').onSnapshot((doc) => { if (doc.exists) { const data = doc.data(); if (data.current_version !== this.currentVersion) { this.notify(data.current_version); } } }); } notify(newVersion) { document.getElementById('update-msg').innerText = `Version ${newVersion} is now live.`; document.getElementById('update-notifier').classList.add('show'); } }
